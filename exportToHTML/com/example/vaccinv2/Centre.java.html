package com.example.vaccinv2;

import com.google.android.gms.maps.model.LatLng;

public class Centre {
    int id;
    String name;
    String lon;
    String lat;

    public Centre(int id, String name, String lon, String lat) {
        this.id = id;
        this.name = name;
        this.lon = lon;
        this.lat = lat;
    }

    public Centre(String name, String lon, String lat) {
        this.name = name;
        this.lon = lon;
        this.lat = lat;
    }

    public Centre(String name, LatLng latLng) {
        this.name = name;
        this.lon = String.valueOf(latLng.longitude);
        this.lat = String.valueOf(latLng.latitude);
    }

    @Override
    public String toString() {
        return "Centre{" +
                "id=" + id +
                ", name='" + name + '\'' +
                ", lon='" + lon + '\'' +
                ", lat='" + lat + '\'' +
                '}';
    }

    public LatLng getLatLng(){
        return new LatLng(Float.parseFloat(this.lat), Float.parseFloat(this.lon));
    }
}
